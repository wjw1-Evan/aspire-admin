# 服务层封装

## 统一使用 request

### **[强制]** 所有 HTTP 请求必须通过 `@umijs/max` 的 `request` 封装。

```tsx
// ✅ 正确：使用统一 request
import { request } from '@umijs/max';

const response = await request<UserResponse>('/api/users', {
  method: 'GET',
  params: { page: 1, pageSize: 20 },
});

// ❌ 禁止：直接 fetch
fetch('/api/users')
  .then(res => res.json())
  .then(data => setUsers(data));
```

## ApiResponse 处理

### 后端响应格式
```typescript
// 后端返回统一 ApiResponse<T>
interface ApiResponse<T> {
  success: boolean;
  data: T;
  errorCode?: string;
  errorMessage?: string;
  showType?: number;
  timestamp: string;
}
```

### 前端处理
```typescript
// ✅ 正确：处理 ApiResponse
import { request } from '@umijs/max';
import type { ApiResponse } from '@/types/unified-api';
import type { UserDto } from './types';

export async function getUserList(): Promise<ApiResponse<UserDto[]>> {
  return request('/api/users/list', {
    method: 'POST',
  });
}

// 使用
const response = await getUserList();
if (response.success) {
  setUsers(response.data);
} else {
  message.error(response.errorMessage);
}

// ❌ 禁止：忽略 success 字段
const response = await getUserList();
setUsers(response.data);  // 未检查 success
```

## Request 拦截器配置

### 错误处理拦截器
```typescript
// src/app.tsx
export const request: RequestConfig = {
  timeout: 30000,
  errorConfig: {
    errorHandler: (error: ResponseError) => {
      if (error.response) {
        const { status } = error.response;
        if (status === 401) {
          // 未授权，跳转登录
          gotoLogin();
        } else if (status === 403) {
          message.error('无权限访问');
        } else {
          message.error(error.message || '请求失败');
        }
      } else {
        message.error('网络错误');
      }
    },
  },
  requestInterceptors: [
    (config) => {
      // 添加 token
      const token = getToken();
      if (token) {
        config.headers = {
          ...config.headers,
          Authorization: `Bearer ${token}`,
        };
      }
      return config;
    },
  ],
  responseInterceptors: [
    (response) => {
      // 统一处理响应
      const data = response.data;
      if (!data.success) {
        message.error(data.errorMessage || '操作失败');
      }
      return response;
    },
  ],
};
```

## API 封装示例

### 基本 CRUD
```typescript
// src/services/task/api.ts
import { request } from '@umijs/max';
import type { ApiResponse } from '@/types/unified-api';
import type { TaskDto, CreateTaskRequest, UpdateTaskRequest } from './types';

// 获取任务列表
export async function queryTasks(params: TaskQueryRequest): Promise<ApiResponse<TaskListResponse>> {
  return request('/api/task/query', {
    method: 'POST',
    data: params,
  });
}

// 获取任务详情
export async function getTaskById(taskId: string): Promise<ApiResponse<TaskDto>> {
  return request(`/api/task/${taskId}`, {
    method: 'GET',
  });
}

// 创建任务
export async function createTask(data: CreateTaskRequest): Promise<ApiResponse<TaskDto>> {
  return request('/api/task/create', {
    method: 'POST',
    data,
  });
}

// 更新任务
export async function updateTask(data: UpdateTaskRequest): Promise<ApiResponse<TaskDto>> {
  return request('/api/task/update', {
    method: 'PUT',
    data,
  });
}

// 删除任务
export async function deleteTask(taskId: string): Promise<ApiResponse<void>> {
  return request(`/api/task/${taskId}`, {
    method: 'DELETE',
  });
}
```

## 常见错误示例

### ❌ 错误 1：未使用统一封装
```tsx
// ❌ 禁止：直接 fetch
useEffect(() => {
  fetch('/api/users')
    .then(res => res.json())
    .then(data => setUsers(data));
}, []);
```

### ❌ 错误 2：未处理错误
```tsx
// ❌ 禁止：未检查 success
const response = await getUserList();
setUsers(response.data);  // 可能失败
```

### ❌ 错误 3：硬编码 URL
```tsx
// ❌ 禁止：硬编码基础 URL
axios.post('http://localhost:5000/api/users', data);

// ✅ 正确：使用相对路径
request('/api/users', { method: 'POST', data });
```
